package com.example.phoneagendacrypto.services

import com.example.phoneagendacrypto.interfaces.IAgendaService
import com.example.phoneagendacrypto.interfaces.IPasswordEncoderService
import com.example.phoneagendacrypto.pojo.Person
import org.springframework.beans.factory.annotation.Autowired
import org.springframework.stereotype.Service

@Service
class EncryptionOrchestratorService {

    @Autowired
    private lateinit var cryptoService: IPasswordEncoderService

    @Autowired
    private lateinit var agendaService: IAgendaService

    fun registerSecurePerson(person: Person): Person {
        // CERINȚĂ BILET: Aplicăm hashing-ul înainte de stocare
        val securedHash = cryptoService.encode(person.accessCode)
        person.accessCode = securedHash
        
        agendaService.addPerson(person)
        return person
    }

    fun verifyAccess(id: Int, plainCode: String): Boolean {
        val person = agendaService.getPerson(id) ?: return false
        // Folosește funcția match pentru validare securizată
        return cryptoService.matches(plainCode, person.accessCode)
    }
}
