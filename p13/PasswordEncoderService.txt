package com.example.phoneagendacrypto.services

import com.example.phoneagendacrypto.interfaces.IPasswordEncoderService
import org.springframework.stereotype.Service
import java.security.MessageDigest

@Service
class PasswordEncoderService : IPasswordEncoderService {

    override fun encode(rawPassword: String): String {
        // Obținem instanța algoritmului cerut în laborator (SHA-256)
        val digest = MessageDigest.getInstance("SHA-256")
        val hashBytes = digest.digest(rawPassword.toByteArray(Charsets.UTF_8))
        
        // Convertim octeții în reprezentare String Hexazecimală
        return hashBytes.joinToString("") { "%02x".format(it) }
    }

    override fun matches(rawPassword: String, encodedPassword: String): Boolean {
        // Verificăm dacă hash-ul parolei introduse coincide cu cel stocat
        return encode(rawPassword) == encodedPassword
    }
}
